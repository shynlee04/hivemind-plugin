/** Search for query in session's child .json files, returning matches. */
async function searchChildJsonFiles(
  projectRoot: string,
  sessionId: string,
  queryLower: string,
): Promise<Array<{ childId: string; field: string; snippet: string }>> {
  const matches: Array<{ childId: string; field: string; snippet: string }> = []
  try {
    const continuityPath = safeSessionPath(projectRoot, sessionId, "session-continuity.json")
    const raw = await readFile(continuityPath, "utf-8")
    const continuity = JSON.parse(raw) as Record<string, unknown>
    const hierarchy = continuity.hierarchy as { children?: Array<{ sessionID: string; childFile: string }> } | undefined
    const children = hierarchy?.children ?? []
    for (const child of children) {
      const childPath = safeSessionPath(projectRoot, sessionId, child.childFile)
      try {
        const childData = JSON.parse(await readFile(childPath, "utf-8")) as Record<string, unknown>
        // Search target fields per D-02
        const fieldsToSearch: Array<{ field: string; extract: (d: Record<string, unknown>) => string | undefined }> = [
          { field: "lastMessage", extract: (d) => typeof d.lastMessage === "string" ? d.lastMessage : undefined },
          { field: "turn.content", extract: (d) => {
            const turns = d.turns as Array<{ content?: string }> | undefined
            return turns ? turns.map((t) => t.content ?? "").join("\n") : undefined
          }},
          { field: "journey[].content", extract: (d) => {
            const journey = d.journey as Array<{ content?: string }> | undefined
            return journey ? journey.map((j) => j.content ?? "").join("\n") : undefined
          }},
          { field: "delegatedBy.subagentType", extract: (d) => {
            const delegatedBy = d.delegatedBy as { subagentType?: string } | undefined
            return delegatedBy?.subagentType
          }},
        ]
        for (const { field, extract } of fieldsToSearch) {
          const value = extract(childData)
          if (value && value.toLowerCase().includes(queryLower)) {
            const truncated = value.length > 200 ? value.slice(0, 200) + "..." : value
            matches.push({ childId: child.sessionID, field, snippet: truncated })
            break  // One match per child file
          }
        }
      } catch { /* skip unreadable child file */ }
    }
  } catch { /* no session-continuity.json found */ }
  return matches
}
```

/** Filter sessions by status, agentType, depth, timeRange. */
async function handleFilterSessions(
  projectRoot: string,
  filters: { status?: string; agentType?: string; minDepth?: number; maxDepth?: number; timeRange?: { after?: string; before?: string }; limit: number },
) {
  try {
    const indexPath = safeSessionPath(projectRoot, "project-continuity", "project-continuity.json")
    const raw = await readFile(indexPath, "utf-8")
    const index = JSON.parse(raw) as {
      sessions?: Record<string, {
        status?: string; childCount?: number; totalDelegationDepth?: number
        created?: string; updated?: string; toolSummary?: Record<string, number>
      }>
      chronologicalOrder?: string[]
    }
    const sessionEntries = Object.entries(index.sessions ?? {})
    
    // Phase 1: index-level filter (fast path — no disk I/O per session)
    let candidates = sessionEntries
    if (filters.status) {
      candidates = candidates.filter(([, meta]) => meta.status === filters.status)
    }
    if (filters.timeRange?.after) {
      const afterMs = new Date(filters.timeRange.after).getTime()
      candidates = candidates.filter(([, meta]) => meta.updated ? new Date(meta.updated).getTime() >= afterMs : true)
    }
    if (filters.timeRange?.before) {
      const beforeMs = new Date(filters.timeRange.before).getTime()
      candidates = candidates.filter(([, meta]) => meta.updated ? new Date(meta.updated).getTime() <= beforeMs : true)
    }
    
    // Phase 2: depth/agentType filter requires reading individual continuity files
    const results: Array<Record<string, unknown>> = []
    const maxScan = Math.min(filters.limit * 3, 100) // Scan more than needed for enrichment
    for (const [sessionId, meta] of candidates.slice(0, maxScan)) {
      const continuityPath = safeSessionPath(projectRoot, sessionId, "session-continuity.json")
      let continuity: Record<string, unknown> | null = null
      try {
        continuity = JSON.parse(await readFile(continuityPath, "utf-8")) as Record<string, unknown>
      } catch { continue }
      
      const delegationDepth = (continuity.delegationDepth as number) ?? 0
      if (filters.minDepth !== undefined && delegationDepth < filters.minDepth) continue
      if (filters.maxDepth !== undefined && delegationDepth > filters.maxDepth) continue
      
      // Agent type filter: check session ID prefix
      const agentType = continuity.sessionID && typeof continuity.sessionID === "string" && (continuity.sessionID as string).startsWith("ses_") ? "session" : "unknown"
      if (filters.agentType && agentType !== filters.agentType) continue
      
      // Rich metadata per D-13
      results.push({
        sessionId,
        agentType,
        depth: delegationDepth,
        status: continuity.status ?? meta.status ?? "unknown",
        lastMessage: typeof continuity.lastMessage === "string"
          ? (continuity.lastMessage as string).slice(0, 500)
          : null,
        createdAt: meta.created ?? null,
        updatedAt: meta.updated ?? null,
        toolSummary: continuity.toolSummary ?? {},
        childCount: continuity.childCount ?? meta.childCount ?? 0,
      })
    }
    
    return renderToolResult(success(`Found ${results.length} sessions matching filters`, {
      totalCandidates: candidates.length,
      results: results.slice(0, filters.limit),
      hasMore: candidates.length > filters.limit,
      filters: {
        status: filters.status ?? null,
        agentType: filters.agentType ?? null,
        minDepth: filters.minDepth ?? null,
        maxDepth: filters.maxDepth ?? null,
        timeRange: filters.timeRange ?? null,
      },
    }))
  } catch (caughtError) {
    return renderToolResult(error(`[Harness] filter-sessions failed: ${caughtError instanceof Error ? caughtError.message : String(caughtError)}`))
  }
}
```
