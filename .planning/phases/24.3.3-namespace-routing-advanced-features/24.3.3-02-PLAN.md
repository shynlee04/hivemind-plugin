    ```typescript
    import { getAppAgents } from "../../shared/app-api.js"
    import type { Agent } from "../../shared/types.js"
    import { createTuiPrompt } from "../../cli/ui/prompts.js"
    
    interface AgentMatchResult {
      agent: string | null
      score: number
      fallback: boolean
      userSpecified: boolean
    }
    
    /**
     * Selects agent based on natural language intent.
     * Pure keyword + fuzzy matching (NO LLM fallback).
     * If no matches → interactive prompt for user to specify exact agent name.
     */
    export async function selectAgent(
      intent: string,
      agents: Agent[] = []
    ): Promise<AgentMatchResult> {
      // Extract keywords from intent
      const keywords = extractKeywords(intent)
      
      // Filter agents by keyword in name/description
      const matches = agents.filter((agent) => {
        const nameMatch = keywords.some((keyword) =>
          agent.name.toLowerCase().includes(keyword)
        )
        const descMatch = agent.description && keywords.some((keyword) =>
          agent.description.toLowerCase().includes(keyword)
        )
        return nameMatch || descMatch
      })
      
      // If no matches → ask user to specify exact name (per D-26)
      if (matches.length === 0) {
        // Get available agent names for prompt
        const agentNames = agents.map(a => a.name).join(", ")
        if (agentNames) {
          const userChoice = await createTuiPrompt({
            type: "text",
            message: "No matching agent found. Please specify exact agent name:",
            initialValue: "",
            hint: `Available agents: ${agentNames}`,
          })
          
          if (userChoice) {
            const specifiedAgent = agents.find(a => a.name.toLowerCase() === userChoice.toLowerCase())
            if (specifiedAgent) {
              return { agent: specifiedAgent.name, score: 1, fallback: false, userSpecified: true }
            }
          }
        }
        return { agent: null, score: 0, fallback: true, userSpecified: false }
      }
      
      // Sort by similarity score (fuzzy matching)
      matches.sort((a, b) => {
        const aScore = calculateSimilarityScore(keywords, a)
        const bScore = calculateSimilarityScore(keywords, b)
        return bScore - aScore
      })
      
      return { agent: matches[0].name, score: matches[0].score, fallback: false, userSpecified: false }
    }
    
    function extractKeywords(text: string): string[] {
      // Simple keyword extraction: split by space, remove special chars
      return text
        .toLowerCase()
        .split(/\s+/)
        .map((word) => word.replace(/[^a-z0-9-]/g, ""))
        .filter((word) => word.length > 0)
    }
    
    function calculateSimilarityScore(keywords: string[], agent: Agent): number {
      let score = 0
      
      // Check name match
      const nameLower = agent.name.toLowerCase()
      keywords.forEach((keyword) => {
        if (nameLower.includes(keyword)) score += 1
      })
      
      // Check description match (if available)
      if (agent.description) {
        const descLower = agent.description.toLowerCase()
        keywords.forEach((keyword) => {
          if (descLower.includes(keyword)) score += 0.5
        })
      }
      
      return score
    }
    ```

    ```typescript
    const discoveryCache = new Map<string, CachedDiscovery>()
    
    interface CachedDiscovery {
      commands: CommandBundle[]
      timestamp: number
      hits: number
      misses: number
    }
    
    const CACHE_TTL_MS = 5000
    
    async function discoverWithCache(projectRoot: string): Promise<CommandBundle[]> {
      const cached = discoveryCache.get(projectRoot)
      if (cached && Date.now() - cached.timestamp < CACHE_TTL_MS) {
        cached.hits++
        logCacheStats() // Shows: hits, misses, total, hit rate
        return cached.commands // Fast path: < 1ms
      }
      
      // Cold path: discover and cache
      const commands = await discoverCommandBundles({ projectRoot })
      discoveryCache.set(projectRoot, { commands, timestamp: Date.now(), hits: 0, misses: 0 })
      
      return commands // ~100ms for cold, < 1ms for warm
    }
    
    function logCacheStats(): void {
      const { hits, misses } = getCacheStats()
      const hitRate = hits + misses > 0 ? (hits / (hits + misses)) * 100 : 0
      console.log(`[Cache] Hits: ${hits}, Misses: ${misses}, Total: ${hits + misses}, Hit Rate: ${hitRate.toFixed(1)}% (target: >50%)`)
    }
    
    export function getCacheStats(): { hits: number; misses: number; total: number } {
      let totalHits = 0
      let totalMisses = 0
      
      for (const cached of discoveryCache.values()) {
        totalHits += cached.hits
        totalMisses += cached.misses
      }
      
      return { hits: totalHits, misses: totalMisses, total: totalHits + totalMisses }
    }
    ```

    ```typescript
    import type { ValidationResult } from "../../shared/types.js"
    
    interface Workflow {
      name: string
      steps: WorkflowStep[]
      triggers?: string[]
    }
    
    interface WorkflowStep {
      id: string
      action: string
      condition?: string
      timeout?: number
    }
    
    /**
     * Parses YAML workflow file.
     * WARNING: Workflow execution deferred to later phase (P24.3.3.2).
     */
    export function parseWorkflowFile(filePath: string): Workflow | null {
      try {
        const content = await fs.promises.readFile(filePath, "utf-8")
        const data = yaml.parse(content) as any
        
        // Validate structure
        if (!data.name || !data.steps || !Array.isArray(data.steps)) {
          return null
        }
        
        return {
          name: data.name,
          steps: data.steps,
          triggers: data.triggers || [],
        }
      } catch (err) {
        return null
      }
    }
    
    /**
     * Validates workflow structure.
     */
    export function validateWorkflow(workflow: Workflow): ValidationResult {
      const errors: string[] = []
      
      if (!workflow.name || workflow.name.length === 0) {
        errors.push("name: required, must be a non-empty string")
      }
      
      if (!workflow.steps || !Array.isArray(workflow.steps) || workflow.steps.length === 0) {
        errors.push("steps: required, must be a non-empty array")
      }
      
      workflow.steps.forEach((step, index) => {
        if (!step.id) {
          errors.push(`steps[${index}].id: required`)
        }
        if (!step.action) {
          errors.push(`steps[${index}].action: required`)
        }
      })
      
      return {
        valid: errors.length === 0,
        errors,
      }
    }
    ```
