    New implementation logic:
    ```typescript
    replayPending(parentSessionId: string): DelegationNotification[] {
      const replayed: DelegationNotification[] = []
      for (const [delegationId, route] of this.routes.entries()) {
        if (route.parentSessionId !== parentSessionId) continue
        
        // Filter notifications: remove expired and exhausted
        const valid: DelegationNotification[] = []
        for (const notification of route.notifications) {
          // We need to check expiration/retry state. Since route.notifications
          // is DelegationNotification[] (not PendingNotificationRecord[]),
          // the retry/expiry state is tracked per-delegation via the
          // internal RouteEntry or a separate tracking structure.
          
          // For now, PendingNotificationRecord carries the retryCount/maxRetries/expiresAt
          // but route.notifications is DelegationNotification[].
          // The retry+TTL filtering should happen at queuePending time (Task 2).
          // At replay time, we just return whatever is in the queue.
          valid.push(notification)
        }
        
        replayed.push(...valid.splice(0))
      }
      return replayed.sort((left, right) => left.timestamp - right.timestamp)
    }
    ```

    **CORRECTION:** The retry/TTL filter should be applied at `queuePending()` time (Task 2), NOT at `replayPending()` time. When a notification expires or exhausts its retries, `queuePending()` should silently drop it — it never reaches `route.notifications`. This means `replayPending()` code stays essentially the same, but `queuePending()` becomes the gate.

    For the retry state tracking, add a `Map<string, { retryCount: number; maxRetries: number; expiresAt: number }>` to the NotificationRouter class to track retry state per delegationId:
    ```typescript
    private readonly retryState = new Map<string, { retryCount: number; maxRetries: number; expiresAt: number }>()
    ```

    Then in `queuePending()`:
    ```typescript
    queuePending(delegationId: string, notification: DelegationNotification): void {
      if (!isParentFacingNotification(notification.type)) return
      const route = this.routes.get(delegationId)
      if (!route) return
      if (notification.idempotencyKey && this.deliveredKeys.has(notification.idempotencyKey)) return
      
      // Initialize or get retry state
      const state = this.retryState.get(delegationId) ?? { retryCount: 0, maxRetries: 1, expiresAt: Date.now() + 5 * 60 * 1000 }
      
      // TTL check
      if (state.expiresAt <= Date.now()) {
        this.retryState.delete(delegationId) // purge expired state
        return // silently drop
      }
      
      // Retry count check
      if (state.retryCount >= state.maxRetries) {
        this.retryState.delete(delegationId) // purge exhausted state
        return // silently drop
      }
      
      // Increment retry count
      state.retryCount++
      this.retryState.set(delegationId, state)
      
      // Persist to route
      if (notification.idempotencyKey) this.deliveredKeys.add(notification.idempotencyKey)
      route.notifications.push(notification)
      if (route.notifications.length > this.pendingLimit)
        route.notifications.splice(0, route.notifications.length - this.pendingLimit)
      this.persistAllPending()
    }
    ```

    deregister(delegationId: string): void {
