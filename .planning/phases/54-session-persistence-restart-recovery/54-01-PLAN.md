# Plan 54-01: Session Persistence + Restart-Recovery

## Objective

Add a new `src/features/tmux/persistence.ts` module (~150 LOC) that persists `TrackedSession` records to `.hivemind/state/tmux-sessions/<sessionId>.json` on every state transition (`active → ready → paused → detached → failed`). On harness startup, `restoreAll()` filters to `paused` and `detached` sessions (skipping `active`/`ready`/`failed`) and returns them sorted by `spawnTime` ascending. The module generates UUIDv7 IDs inline using `node:crypto.getRandomValues()` (no new `package.json` deps — P20 invariant) and mirrors the P53 D-04 silent-fallback contract: all errors are logged via `logWarn`, no throw crosses the module boundary.

The 27-tool-key invariant is preserved — `SessionManagerAdapter` in `types.ts` remains byte-identical. The single 5-task plan (1 wave, ≤ 30 LOC net additive to `session-manager.ts`) keeps the EXECUTE phase within a single context window.

**UUIDv7 inline generator (REQ-54-04, D-54-07 — 20-line function using `getRandomValues`):**

```typescript
function generateUuidV7(): string {
  // 48-bit big-endian ms timestamp (12 hex chars)
  const tsHex = Date.now().toString(16).padStart(12, "0")
  // 80 random bits (10 bytes) — Node 20+ CSPRNG
  const rand = new Uint8Array(10)
  getRandomValues(rand)
  // Set version bits 48-51 = 0b0111 (version 7) → byte 6 high nibble = 7
  rand[6] = ((rand[6] ?? 0) & 0x0f) | 0x70
  // Set variant bits 64-65 = 0b10 (RFC 4122) → byte 8 high 2 bits = 10
  rand[8] = ((rand[8] ?? 0) & 0x3f) | 0x80
  const hex = Array.from(rand, (b) => b.toString(16).padStart(2, "0")).join("")
  // Format: xxxxxxxx-xxxx-7xxx-yxxx-xxxxxxxxxxxx
  return `${tsHex.slice(0, 8)}-${tsHex.slice(8, 12)}-${hex.slice(0, 3)}7${hex.slice(3, 6)}-${(rand[8] ?? 0).toString(16).padStart(2, "0")}${hex.slice(9, 12)}-${hex.slice(12, 20)}`
}
```

**Atomic write helper (REQ-54-02, D-54-10 — `wx` flag + EEXIST → `w` fallback):**

```typescript
async function writeRecord(filePath: string, record: PersistedSession): Promise<void> {
  const content = JSON.stringify(record, null, 2)
  try {
    await writeFile(filePath, content, { encoding: "utf-8", flag: "wx" })
  } catch (err) {
    if (isEEXIST(err)) {
      // EEXIST: file already exists from prior persist → overwrite with latest
      await writeFile(filePath, content, { encoding: "utf-8", flag: "w" })
      return
    }
    throw err
  }
}

function isEEXIST(err: unknown): boolean {
  return typeof err === "object" && err !== null && "code" in err && (err as { code: string }).code === "EEXIST"
}
```

**`persist()` (REQ-54-02 acceptance — 5 sequential state transitions persist correctly, lastTransitionAt monotonic):**

async function persist(record: PersistedSession): Promise<void> {
  try {
    record.lastTransitionAt = record.lastTransitionAt ?? Date.now()
    record.schemaVersion = 1
    if (!isValidSessionState(record.state)) {
      logWarn(`persist: invalid state literal: ${String(record.state)} for sessionId=${record.sessionId}`)
      return  // D-04 silent-fallback
    }
    const filePath = join(stateRoot, `${record.sessionId}.json`)
    await writeRecord(filePath, record)
  } catch (err) {
    logWarn(`persist failed for sessionId=${record.sessionId} state=${record.state}`, err)
  }
}
```

const ALIVE_STATES: ReadonlySet<SessionState> = new Set(["paused", "detached"])

async function restoreAll(): Promise<PersistedSession[]> {
  let files: string[]
  try {
    files = await readdir(stateRoot)
  } catch {
    return []  // Fresh project — empty stateRoot, no throw (D-04)
  }
  const records: PersistedSession[] = []
  for (const file of files) {
    if (!file.endsWith(".json")) continue
    try {
      const raw = await readFile(join(stateRoot, file), "utf-8")
      const parsed = JSON.parse(raw) as PersistedSession
      if (!isValidPersistedSession(parsed)) {
        logWarn(`restoreAll: skipping malformed record (missing fields): ${file}`)
        continue
      }
      if (ALIVE_STATES.has(parsed.state)) {
        records.push(parsed)
      }
    } catch (err) {
      logWarn(`restoreAll: skipping unreadable/malformed JSON: ${file}`, err)
    }
  }
  records.sort((a, b) => a.spawnTime - b.spawnTime)
  return records
}
```
