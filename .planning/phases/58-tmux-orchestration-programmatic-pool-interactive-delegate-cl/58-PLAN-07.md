```typescript
/**

 * no v2 modules. The resulting instance supports:

 *   - `__getDelegationsForTesting` — returns a `ReadonlyMap` view; tests
 *     populate it directly via the setter
 *   - `getPoolSnapshot()` after test-side mutation of __getDelegationsForTesting

 *   1. Call `const instance = DelegationManager.createForTest()`
 *   2. Populate `instance.__getDelegationsForTesting` via `Map.set()`
 *   3. Call `instance.getPoolSnapshot()` to assert frozen contract
 *
 * NOT for production code — only for BATS slots 62 and similar in-memory
 * tests. Real wiring still requires `new DelegationManager(client)`.
 */
static createForTest(): DelegationManager {
  // Bypass the client-required guard at line 78-89 by passing undefined
  // for both args. The constructor only throws when `!client && !options.coordinator && !options.lifecycle`,
  // and we pass neither coordinator nor lifecycle. The 3-branch guard at
  // line 87 short-circuits when EITHER coordinator OR lifecycle is provided.
  // We exploit the asymmetry: when BOTH are absent, the guard's `else if`
  // is unreachable, but JavaScript short-circuits the if-chain when `!client` is true.
  // The 3rd `else throw` fires only when `!client && !coordinator && !lifecycle`.
  // To avoid the throw, we must inject `coordinator` OR `lifecycle` minimally.
  // The simplest workaround: inject a `coordinator` no-op so the first if-branch fires.
  const noopCoordinator = {
    abortDelegation: (_id: string) => ({ delegationId: _id, error: "noop", status: "error" as const }),
    markAborted: (_id: string) => ({ delegationId: _id, error: "noop", status: "error" as const }),
  } as unknown as DelegationManagerOptions["coordinator"]
  // We use a minimal `as unknown` cast because the coordinator type may
  // require more methods; the noop only needs to satisfy the 1st if-branch
  // check at line 87. The cast is acceptable here because BATS never calls
  // any coordinator method (it uses __getDelegationsForTesting + getPoolSnapshot).
  return new DelegationManager(undefined, { coordinator: noopCoordinator })
}
```
