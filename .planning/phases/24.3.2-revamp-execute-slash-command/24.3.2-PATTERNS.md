```typescript
async function discoverCommandBundles(projectRoot: string): Promise<CommandBundle[]> {
  const commandsDir = path.join(projectRoot, ".opencode", "commands")
  try {
    await access(commandsDir)
  } catch {
    return []
  }
  const files = await readdir(commandsDir)
  const bundles: CommandBundle[] = []
  for (const file of files) {
    if (!file.endsWith(".md")) continue
    const content = await readFile(path.join(commandsDir, file), "utf-8")
    const parsed = parseCommandFrontmatter(content)
    if (parsed) bundles.push(parsed)
  }
  return bundles
}
```

```typescript
export class DelegationManager {
  private runtime: RuntimeDelegationManager
  private completionDetector: CompletionDetector
  private ptyManager: PtyManager
  private coordinator: DelegationCoordinator

  constructor(opts: DelegationManagerOptions) {
    this.runtime = opts.runtime
    this.completionDetector = opts.completionDetector
    this.ptyManager = opts.ptyManager
    this.coordinator = opts.coordinator
  }

  async delegate(req: DelegationSpawnRequest): Promise<DelegationResult> {
    // Route to appropriate sub-manager based on execution mode
    switch (req.executionMode) {
      case "pty": return this.ptyManager.delegate(req)
      case "headless": return this.coordinator.delegate(req)
      default: return this.runtime.delegate(req)
    }
  }
}
```

### Error Handling (never throw)
**Source:** `src/tools/delegation/delegate-task.ts` (lines 72–80), `src/shared/tool-response.ts`, `src/shared/helpers.ts` (line 293)
**Apply to:** All tool.execute functions

```typescript
// In the outer catch block of tool.execute:

  // ... dispatch logic

  return error({
    metadata: { error: true, command: input.command },
```

- Never `throw` inside `tool.execute` — always return `error()` envelope
