```typescript
export enum EvidenceLane {
  LOCAL_DIAGNOSTICS = 'local_diagnostics',      // VER-01
  INTEGRATION_CHECKS = 'integration_checks',    // VER-02
  LIVE_OFFICIAL_INTERFACE_PROOF = 'live_official_interface_proof'  // VER-03
}

export type EvidenceStatus = 'pass' | 'fail' | 'unavailable' | 'not_applicable';

export interface EvidenceResult<T = unknown> {
  lane: EvidenceLane;
  status: EvidenceStatus;
  label: '[non-live evidence]' | null;
  justification: string | null;
  evidence: T;
}

export function createEvidenceResult<T>(
  lane: EvidenceLane,
  status: EvidenceStatus,
  evidence: T,
  opts: { label?: '[non-live evidence]' | null; justification?: string | null } = {}
): EvidenceResult<T> {
  return {
    lane,
    status,
    label: opts.label ?? null,
    justification: opts.justification ?? null,
    evidence
  };
}
```

This file is the SINGLE SOURCE OF TRUTH for evidence lane types. All validators (04-02, 04-03) must import from this file, not define their own lane types.</action>
