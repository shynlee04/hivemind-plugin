       import { getSessionContinuity } from "../../continuity.js"
       import type { DelegationCompletionSnapshot, CompletionStatus } from "./types.js"

       export class ParentCoordinator {
         private childSessionIDs: Set<string> = new Set()

         /** Register a child delegation session for tracking. */
         registerChild(childSessionID: string): void

         /** Remove a child from tracking (cleanup). */
         unregisterChild(childSessionID: string): void

         /** Query continuity store for all tracked children's lifecycle phases. */
         snapshotDelegationStatus(): DelegationCompletionSnapshot

         /** True when every tracked child has completed OR failed. */
         allDelegationsComplete(): boolean

          /** True when delegations done AND parent state says no further auto-loop/final-synthesis work remains. */
          canCloseMainSession(mainSessionID: string): boolean
       }
       ```

       Implementation details:
       - `snapshotDelegationStatus()` iterates `childSessionIDs`, calls `getSessionContinuity(id)` for each, reads `metadata.lifecycle?.phase`

       - Everything else (running, queued, dispatching, created, or no record) goes into active set

       // Parent coordination check
        const children = stateManager.getSubagents(sessionID)
        if (children.size > 0 && !continuity.metadata.delegationPacket) {
         // This is a PARENT session with active delegations
         // Check if all delegations are complete via ParentCoordinator
         const coordinator = parentCoordinators.get(sessionID) ?? new ParentCoordinator()
         for (const childID of children) coordinator.registerChild(childID)

         if (!coordinator.allDelegationsComplete()) {
           // Auto-loop: send prompt asking LLM to wait for delegations
           const state = getParentAutoLoopState(parentAutoLoopStates, sessionID)
           if (state.iterations >= parentLoopConfig.maxIterations) return // exhausted
           state.iterations += 1
           await lifecycleManager.requestAutoLoopRetry({
             sessionID,
             promptText: buildParentAutoLoopPrompt(coordinator.snapshotDelegationStatus(), state),
           })
         } else {
           // All complete — clean up coordinator
           parentCoordinators.delete(sessionID)
           // Send synthesis prompt
           await lifecycleManager.requestAutoLoopRetry({
             sessionID,
             promptText: buildParentSynthesisPrompt(coordinator.snapshotDelegationStatus()),
           })
         }
       }
       ```

       The `buildParentAutoLoopPrompt()` function builds a system message:
       ```
       <system_reminder>Parent auto-loop poll ${iterations}/${maxIterations}</system_reminder>
       Delegation status: X active, Y completed, Z failed.
       ${details of each delegation}
       Continue waiting for background delegations to complete.
       When all are done, synthesize results and emit <promise>DONE</promise>.
       ```

       The `buildParentSynthesisPrompt()` function builds:
       ```
       <system_reminder>All delegations complete</system_reminder>
       All background tasks have finished. Synthesize their results now.
       Delegation summary: ${details}
       ```

       **CRITICAL:** The parent auto-loop is SEPARATE from the child auto-loop. They coexist in the same event handler but branch on different conditions:
       - Child auto-loop: `continuity.metadata.delegationPacket` exists → child session waiting for `<promise>DONE</promise>`
        - Parent auto-loop: `stateManager.getSubagents(sessionID)` has entries → parent session waiting for delegation completion

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Continuity store → parent-coordinator | Delegation phase data read from JSON persistence. Stale or corrupted records could produce false "complete" signals. |
| Auto-loop prompt → LLM | Auto-loop injected prompts become part of the LLM context. Malicious delegation metadata could inject content. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09.2-04 | Denial of Service | parent-coordinator | mitigate | Max 10 auto-loop iterations prevents infinite context consumption. Hard cap enforced in getAutoLoopState. |
| T-09.2-05 | Tampering | parent-auto-loop prompt | mitigate | Auto-loop prompt is system_reminder tagged, built from continuity data (trusted), not user input. Delegation metadata fields are read-only from continuity store. |
| T-09.2-06 | Denial of Service | failure-handler | mitigate | Hard limit of 2 retries (FAILURE_RETRY_LIMIT=2). No infinite retry loops possible. isPermanentlyFailed() returns true after exhaustion. |
| T-09.2-07 | Information Disclosure | parent-auto-loop | accept | Auto-loop prompt includes delegation session IDs and status — this is internal harness data, not user-visible. Low risk. |
</threat_model>
