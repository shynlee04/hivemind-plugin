STEP 1 — Create `src/tools/delegate-task.ts`:

Follow the project's established tool pattern (using `tool()` from `@opencode-ai/plugin` and `z` from `zod`):

```typescript
import { tool } from "@opencode-ai/plugin"
import { z } from "zod"
import type { DelegationManager } from "../lib/delegation-manager.js"

export function createDelegateTaskTool(delegationManager: DelegationManager) {
  return tool({
    name: "delegate-task",
    description: "Delegate work to a specialist agent. Creates a child session, dispatches the task, and returns the result. Use sync mode (default) for immediate results, or async mode for long-running tasks.",
    parameters: z.object({
      agent: z.string().describe("Agent name to delegate to (must be in VALID_AGENTS)"),
      prompt: z.string().describe("The task prompt to send to the agent"),
      title: z.string().optional().describe("Optional title for the child session"),
      async: z.boolean().default(false).describe("If true, return delegation ID immediately without waiting for completion"),
      timeoutMs: z.number().min(1000).max(1800000).optional().describe("Timeout in milliseconds (default: 15 minutes, max: 30 minutes)"),
    }),
    execute: async (params, { session }) => {
      const parentSessionId = session.id
      try {
        if (params.async) {
          const { delegationId } = await delegationManager.delegateAsync({
            parentSessionId,
            agent: params.agent,
            prompt: params.prompt,
            title: params.title,
            timeoutMs: params.timeoutMs,
          })
          return {
            status: "dispatched",
            delegationId,
            message: `Async delegation dispatched to ${params.agent}. Use delegation ID to check status.`,
          }
        } else {
          const result = await delegationManager.delegateSync({
            parentSessionId,
            agent: params.agent,
            prompt: params.prompt,
            title: params.title,
            timeoutMs: params.timeoutMs,
          })
          return result
        }
      } catch (err) {
        const message = err instanceof Error ? err.message : String(err)
        return { status: "error", error: message }
      }
    },
  })
}
```

The `session.id` comes from the tool execution context provided by OpenCode — it's the parent session ID. If the SDK context doesn't provide `session.id`, read it from `process.env.OPENCODE_SESSION_ID` as fallback.
