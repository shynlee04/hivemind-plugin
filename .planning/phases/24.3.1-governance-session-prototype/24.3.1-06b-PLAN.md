          prompt: args.brief,
          queueKey: `agent:${resolvedAgent}`,
          surface: "governance-dispatch",

            sessionID,
            title: sessionTitle,
          }),

      }

      try {
        await sendPrompt(client, sessionID, { parts: [{ type: "text", text: args.brief }] })
