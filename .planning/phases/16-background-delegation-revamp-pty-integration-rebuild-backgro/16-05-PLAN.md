    ```typescript
    private async performStabilityPoll(delegationId: string): Promise<void> {
      const delegation = this.delegations.get(delegationId)
      if (!delegation || delegation.status !== "running") {
        return
      }

      // Fetch actual message count from child session
      const currentMessageCount = await this.fetchChildMessageCount(delegation.childSessionId)

      if (currentMessageCount !== delegation.lastMessageCount) {
        // Messages still arriving — reset stability counter
        delegation.lastMessageCount = currentMessageCount
        delegation.stablePollCount = 0
        this.persistAllDelegations()
      } else {
        // Message count unchanged — increment stability counter
        delegation.stablePollCount += 1
        this.persistAllDelegations()
      }

      if (delegation.stablePollCount >= STABILITY_THRESHOLD) {
        // True stability confirmed — message count unchanged for N consecutive polls
        await this.finalizeDelegation(delegationId)
        return
      }

      // Not yet stable — schedule next poll
      if (!this.stabilityTimers.has(delegationId)) {
        this.scheduleStabilityPoll(delegationId)
      }
    }
    ```

    ```typescript
    private async fetchChildMessageCount(sessionId: string): Promise<number> {
      try {
        const messages = unwrapData<MessageLike[]>(

        return messages?.length ?? 0

        // If fetch fails, return current count to avoid false stability

      }
    }
    ```
