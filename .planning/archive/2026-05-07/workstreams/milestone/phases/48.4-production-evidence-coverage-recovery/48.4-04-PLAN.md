```typescript
function createStatefulFakeClient() {
  const sessions = new Map<string, { id: string; status: string; messages: unknown[] }>()
  return {
    session: {
      create: vi.fn(async (opts: unknown) => {
        const id = `fake-ses-${sessions.size + 1}`
        sessions.set(id, { id, status: "running", messages: [] })
        return { data: { id } }
      }),
      status: vi.fn(async (opts: { sessionID: string }) => {
        const s = sessions.get(opts.sessionID)
        return { data: s ? { status: { type: s.status } } : {} }
      }),
      messages: vi.fn(async (opts: { sessionID: string }) => {
        const s = sessions.get(opts.sessionID)
        return { data: s?.messages ?? [] }
      }),
      prompt: vi.fn(async (sessionID: string, body: unknown) => {
        const s = sessions.get(sessionID)
        if (s) { s.status = "completed"; s.messages.push({ role: "assistant", content: body }) }
        return {}
      }),
      abort: vi.fn(async (sessionID: string) => {
        const s = sessions.get(sessionID)
        if (s) { s.status = "cancelled" }
        return {}
      }),
    },
    // Helper: mutate session state from tests
    _setStatus(sessionID: string, status: string) {
      const s = sessions.get(sessionID)
      if (s) s.status = status
    },
    _getSession(sessionID: string) {
      return sessions.get(sessionID)
    },
    _getAllSessions() {
      return [...sessions.values()]
    },
  }
}
```

```typescript
import { delegateTaskTool } from "../../src/tools/delegate-task.js"

describe("contract-based tests", () => {
  let tool: ReturnType<typeof delegateTaskTool>
  let fakeManager: { delegations: Map<string, unknown> }

  beforeEach(() => {
    fakeManager = {
      delegations: new Map(),
      dispatchDelegation: vi.fn(async (opts: { agent: string; prompt: string }) => {
        const id = `delg_${fakeManager.delegations.size + 1}`
        fakeManager.delegations.set(id, { id, agent: opts.agent, status: "dispatched" })
        return { delegationId: id, status: "dispatched" }
      }),
    }
    tool = delegateTaskTool({
      client: createStatefulFakeClient() as never,
      delegationManager: fakeManager as never,
    })
  })

  it("rejects empty agent name", async () => {
    const result = await tool.execute({ agent: "", prompt: "do work" })
    expect(result.status).toBe("error")
  })

  it("accepts valid input and returns standard envelope", async () => {
    const result = await tool.execute({ agent: "builder", prompt: "build component" })
    expect(result).toMatchObject({
      status: "ok",
      delegationId: expect.any(String),
    })
  })

  it("creates delegation record on success", async () => {
    await tool.execute({ agent: "builder", prompt: "build component" })
    expect(fakeManager.delegations.size).toBe(1)
  })
})
```
